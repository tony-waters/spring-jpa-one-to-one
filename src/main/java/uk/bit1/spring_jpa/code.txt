@NoArgsConstructor(access = AccessLevel.PROTECTED)
public class ProfileB {

    @Id
    @GeneratedValue(strategy= GenerationType.AUTO)
    @Getter
    private Long id;

    @OneToOne(
            cascade = CascadeType.ALL,
            orphanRemoval = true
    )
    @JoinColumn(name = "customer_id") // Owner side is here
    private CustomerB customer;

    @Getter

        this.customer = customer;
    }

    void clearCustomerInternal() {
        this.customer = null;
    }

}

@NoArgsConstructor(access = AccessLevel.PROTECTED)
public class CustomerB {

    @Id
    @GeneratedValue(strategy=GenerationType.AUTO)
    @Getter
    private Long id;

    // Customer is the Inverse side
    // Customer is the Parent side

        this.profile = new ProfileB(marketingOptIn);
        profile.setCustomerInternal(this);
        return profile;
    }

    public void removeProfile() {
        this.profile = null;
    }

}
